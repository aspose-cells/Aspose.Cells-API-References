### Örnekler

```csharp

[C#]

//Bir Çalışma Kitabı nesnesini başlatma
Workbook workbook = new Workbook();
//Workbook nesnesine yeni bir çalışma sayfası ekleme
workbook.Worksheets.Add();
//Yeni eklenen çalışma sayfasının sayfa indeksini geçerek referansının alınması
Worksheet worksheet = workbook.Worksheets[0];
//"A1" hücresindeki bir URL'ye köprü ekleme
int index = worksheet.Hyperlinks.Add("A1", 1, 1, "http://www.aspose.com");
// Dizine göre bir Köprü alma.
Hyperlink hyperlink = worksheet.Hyperlinks[index];
//Bu köprünün görüntülenen metni ayarlanıyor.
hyperlink.TextToDisplay = "Aspose";
//Excel dosyasını kaydetme
workbook.Save("book1.xls");
```
