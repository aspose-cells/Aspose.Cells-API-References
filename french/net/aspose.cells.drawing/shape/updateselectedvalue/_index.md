```csharp
public void UpdateSelectedValue()
```

### Exemples

```csharp

[C#]
// Initialise un nouveau classeur.
//Classeur livre = nouveau Classeur();

Cell cell = null;
for (int i = 0; i< 10; ++i)
{
    cell = book.Worksheets[0].Cells[i, 0];
    cell.Value = i + 1;
}
   
//Créer un objet ListBox

//Contrôles ActiveX
//Aspose.Cells.Drawing.Shape listBox = book.Worksheets[0].Shapes.AddActiveXControl( Aspose.Cells.Drawing.ActiveXControls.ControlType.ListBox,2, 0, 2, 0, 130, 130);

//Contrôles de formulaire
Aspose.Cells.Drawing.Shape listBox = book.Worksheets[0].Shapes.AddListBox(2, 0, 2, 0, 130, 130);

// Définit la plage utilisée pour remplir le contrôle.
listBox.SetInputRange("$A$1:$A$6", false, false);

//Définit la plage liée à la valeur du champ.
listBox.SetLinkedCell("$A$12", false, true);

ListBox listbx = (ListBox)listBox;

//Définir la valeur de la cellule A12
cell = book.Worksheets[0].Cells[11, 0];
cell.Value = 3;

//Mise à jour la valeur sélectionnée par la valeur de la cellule liée.
listBox.UpdateSelectedValue();

//-1 par défaut, aucune option sélectionnée
if(listbx.IsSelected(2))
{
    //L'option 3 de la ListBox est sélectionnée
}

//Modifier la valeur d'une cellule liée
cell.Value = 4;

//Mise à jour la valeur sélectionnée par la valeur de la cellule liée.
listBox.UpdateSelectedValue();
if(listbx.IsSelected(3))
{
    //L'option 4 de la ListBox est sélectionnée
}
```
