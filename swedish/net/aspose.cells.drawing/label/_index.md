### Exempel

```csharp
//Skapa en ny arbetsbok.
Workbook workbook = new Workbook();
 
//Hämta det första kalkylbladet i arbetsboken.
Worksheet sheet = workbook.Worksheets[0];
 
//Lägg till en ny etikett i kalkylbladet.
Aspose.Cells.Drawing.Label label = sheet.Shapes.AddLabel(2, 0, 2, 0, 60, 120);
 
//Ställ in rubriken för etiketten.
label.Text = "This is a Label";
 
//Ange placeringstyp, så som
//etiketten är fäst vid cellerna.
label.Placement = PlacementType.FreeFloating;
 
//Ställ in fyllningsfärgen för etiketten.
label.Fill.FillType = FillType.Solid;
label.Fill.SolidFill.Color = Color.Yellow;
 
//Sparar filen.
workbook.Save("tstlabel.xls");
```
