// إنشاء كائن مصنف

int chartIndex = worksheet.Charts.Add(ChartType.Column, 5, 0, 15, 5);
// الوصول إلى مثيل المخطط المضاف حديثًا
Chart chart = worksheet.Charts[chartIndex];
// إضافة NSeries (مصدر بيانات المخطط) إلى المخطط الذي يتراوح من خلية "A1" إلى "B4"

// تعيين مصدر البيانات لبيانات فئة سلسلة NS
chart.NSeries.CategoryData = "C1:C4";
// ضبط وحدة عرض محور القيمة (ص).
chart.ValueAxis.DisplayUnit = DisplayUnitType.Hundreds;
DisplayUnitLabel displayUnitLabel = chart.ValueAxis.DisplayUnitLabel;
// تعيين تسمية وحدة العرض المخصصة
displayUnitLabel.Text = "100";
// حفظ ملف Excel
workbook.Save("book1.xls");
