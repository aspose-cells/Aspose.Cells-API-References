namespace AsposeCellsExamples
{
    using Aspose.Cells;
    using Aspose.Cells.Metas;
    using System;

    public class SensitivityLabelPropertyIdDemo
    {

        {
            // Create a new workbook
            Workbook workbook = new Workbook();

            {
