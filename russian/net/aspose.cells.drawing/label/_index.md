//Создать новую книгу.
Workbook workbook = new Workbook();

Worksheet sheet = workbook.Worksheets[0];
 
//Добавить новую метку на лист.
Aspose.Cells.Drawing.Label label = sheet.Shapes.AddLabel(2, 0, 2, 0, 60, 120);
 
//Установить заголовок метки.
label.Text = "This is a Label";
 
//Установите тип размещения, как
//метка прикрепляется к ячейкам.

//Задаем цвет заливки метки.
label.Fill.FillType = FillType.Solid;
label.Fill.SolidFill.Color = Color.Yellow;
 
// Сохраняет файл.
workbook.Save("tstlabel.xls");
